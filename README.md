
# Install DNS SERVER

apt update -y && apt install sudo && sudo apt install -y wget && sudo apt install -y bzip2 gzip coreutils screen curl && wget https://raw.githubusercontent.com/vpnlegasi/addip-dns/main/install-dns.sh && chmod +x install-dns.sh && ./install-dns.sh
